```javascript name=gemini.js
const genAI = new GoogleGenerativeAI("YOUR_FREE_API_KEY");

}

  const result = await model.generateContent([
    question,
    { inlineData: { data: imageBase64, mimeType: "image/jpeg" } }
  ]);
  return result.response.text();
}
```

### Groq Setup (for chat & quizzes):
```javascript name=groq.js
import Groq from "groq-sdk";

const groq = new Groq({ apiKey: "YOUR_FREE_API_KEY" });

async function askSciBot(question) {
  const response = await groq.chat.completions.create({
    messages: [
      { 
        role: "system", 
        content: "You are SciBot, a friendly AI science tutor for high school students. Explain concepts simply and encourage learning."
      },
      { role: "user", content: question }
    ],
    model: "llama3-70b-8192",
  });
  return response.choices[0].message.content;
}
```
